# Rich Text Editor

The Rich Text Editor component provides a user-friendly interface for creating and editing formatted text content. It allows users to apply various formatting options, insert media, and structure content with headings, lists, and other elements.

## Usage

<Canvas of={RichTextEditorStories.Documentation} source={ { code: `
import React, { useState, useEffect } from "react";
import { RichTextEditor, type RichTextEditorValue } from "@webiny/admin-ui";

const RichTextEditorExample = () => {
    const [content, setContent] = useState<RichTextEditorValue | undefined>(undefined);
    const [validation, setValidation] = useState({ isValid: true, message: "" });

    const handleChange = (newContent: RichTextEditorValue) => {
        setContent(newContent);

        // Simple required validation
        if (!newContent || newContent.length === 0) {
            setValidation({ isValid: false, message: "This field is required" });
        } else {
            setValidation({ isValid: true, message: "" });
        }
    };

    return (
        <RichTextEditor
            label="Content Editor"
            description="Enter the content for your page or post"
            note="Use the toolbar to format your text and add media"
            required={true}
            value={content}
            onChange={handleChange}
            validation={validation}
        />
    );
};

export default RichTextEditorExample;
